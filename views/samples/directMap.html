<div class="container">
    <form class="directions-demo">
        <div class="input-group">
            <label for="origin">Origin:</label>
            <!-- An input box will be needed for the starting point
                    We make the input box an agg-auto-complete box and
                    set the model to dir.request.origin -->
            <input agg-auto-complete="dir.request.origin" id="origin" type="text" ng-model="dir.input.origin">
        </div>
        <div class="input-group">
            <label for="destination">Destination:</label>
            <!-- An input box will also be needed for the destination
                    We also make it an agg-auto-complete box and set
                    the model to dir.request.destination -->
            <input agg-auto-complete="dir.request.destination" id="destination" type="text" ng-model="dir.input.destination">
        </div>

        <!-- We want to allow the user to select a travel mode so we create some radio buttons
                and add the ng-model directive to them with the model dir.request.travelMode. The
                possible values are WALKING, DRIVING, BICYCLING, and TRANSIT -->
        <h5>Travel Mode</h5>
        <div class="travel-mode">
            <div class="col-xs-3">
                <label for="driving">Driving</label>
                <input id="driving" type="radio" name="mode" ng-model="dir.request.travelMode" value="DRIVING" checked>
            </div>
            <div class="col-xs-3">
                <label for="walking">Walking</label>
                <input id="walking" type="radio" name="mode" ng-model="dir.request.travelMode" value="WALKING">
            </div>
            <div class="col-xs-3">
                <label for="bicycling">Bicycling</label>
                <input id="bicycling" type="radio" name="mode" ng-model="dir.request.travelMode" value="BICYCLING">
            </div>
            <div class="col-xs-3">
                <label for="transit">Transit</label>
                <input id="transit" type="radio" name="mode" ng-model="dir.request.travelMode" value="TRANSIT">
            </div>
        </div>

        <!-- This button triggers the clear() method which clears the input and result models -->
        <button class="btn btn-danger" ng-click="dir.clear()">Clear</button>
    </form>
</div>
